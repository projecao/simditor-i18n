## Install 

`npm install --save simditor-i18n`

## Usage

```js
  import Simditor from 'simditor'
  
  import 'simditor-i18n'
  
  Simditor.locale = 'pt-BR' // i18n
  
  var editor = new Simditor({
    textarea: $('#editor')
    //optional options
  });
```
